    data class Pedido(
    val id: Int,
    val status: StatusPedido
)

enum class StatusPedido {
    NOVO,
    PROCESSADO,
    DESCONHECIDO
}

interface ProcessadorPedido {
    fun processar(pedido: Pedido)
}

class NovoPedido : ProcessadorPedido {
    override fun processar(pedido: Pedido) {
        println("Pedido em processamento: ${pedido.id}")
    }
}

class PedidoProcessado : ProcessadorPedido {
    override fun processar(pedido: Pedido) {
        println("Pedido já processado: ${pedido.id}")
    }
}

class PedidoProcessor {

    fun processarPedido(pedido: Pedido) {

        when (pedido.status) {
            StatusPedido.NOVO -> NovoPedido().processar(pedido)
            StatusPedido.PROCESSADO -> PedidoProcessado().processar(pedido)
            else -> println("Status desconhecido")
        }
    }
}
